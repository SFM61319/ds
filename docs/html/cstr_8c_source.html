17ds_result_t
18ds_cstr_allocate (ds_cstr_t *const self, ds_usize_t const size)
19{
20 DS_RESULT_PROPAGATE_IF_NULL (self);
21 if (size == DS_USIZE_MIN)
22 {
23 // We need not allocate memory since the requested @p size is `0`. Hence,
24 // we simply point to `NULL` to indicate the skipped allocation.
25 *self = NULL;
26 }
27 else
28 {
29 *self = (ds_cstr_t)ds_allocator_global.allocate (DS_CHAR_BYTES * size);
30 if (ds_helpers_is_null (*self))
31 {
32 return DS_RESULT_ERR_MEM_ALLOC_FAILED;
33 }
34 }
35
36 return DS_RESULT_OK;
37}

39ds_result_t
40ds_cstr_reallocate (ds_cstr_t *const src_cstr_ptr, ds_usize_t const src_size,
41 ds_cstr_t *const dst_cstr_ptr, ds_usize_t const dst_size)
42{
43 DS_RESULT_PROPAGATE_IF_NULL (src_cstr_ptr);
44 DS_RESULT_PROPAGATE_IF_NULL (dst_cstr_ptr);
45
46 ds_cstr_t const src_str = *src_cstr_ptr;
47
48 // The source is `NULL`, so freshly allocate the destination string instead.
49 if (ds_helpers_is_null (src_str))
50 {
51 return ds_cstr_allocate (dst_cstr_ptr, dst_size);
52 }
53
54 // Move the string from the source to the destination in case reallocation is
55 // not necessary or fails.
56 *src_cstr_ptr = NULL;
57 *dst_cstr_ptr = src_str;
58
59 // We need not reallocate the string since the current size and the required
60 // size are the same. So we leave the moved string as is and return Ok.
61 if (dst_size == src_size)
62 {
63 return DS_RESULT_OK;
64 }
65
66 ds_cstr_t const dst_str = (ds_cstr_t)ds_allocator_global.reallocate (
67 (ds_cstr_mut_t)src_str, DS_CHAR_BYTES * dst_size);
68
69 // Reallocation failed, leave the moved string as is and return the error.
70 if (ds_helpers_is_null (dst_str))
71 {
72 return DS_RESULT_ERR_MEM_ALLOC_FAILED;
73 }
74
75 // Reallocation was successful, point to the new destination string instead.
76 *dst_cstr_ptr = dst_str;
77 return DS_RESULT_OK;
78}

80ds_result_t
81ds_cstr_deallocate (ds_cstr_t *const self)
82{
83 DS_RESULT_PROPAGATE_IF_NULL (self);
84 ds_allocator_global.deallocate ((ds_cstr_mut_t)*self);
85
86 // Point to `NULL` to avoid memory leaks or other critical issues caused when
87 // deallocating a pointer twice without any allocation in between, as `NULL`
88 // pointers are simply ignored by deallocating functions.
89 *self = NULL;
90 return DS_RESULT_OK;
91}
